  - name: planRequired
    displayName: Plan Required
    type: string
    values:
      - True
      - False

  - name: applyRequired
    displayName: Apply Required
    type: string
    values:
      - True
      - False

  - name: scanRequired
    displayName: Scan Required
    type: string
    values:
      - True
      - False

variables: 
  work_dir: '$(System.DefaultWorkingDirectory)/environment/dev'
  service_conn: 'PrabhaSC-WIF'

jobs:
  - job: initPlanJob
    displayName: Terraform Init & Plan
    steps:
    - task: TerraformInstaller@1
      displayName: Terraform Install
      inputs:
        terraformVersion: 'latest'

      inputs:
        provider: 'azurerm'
        command: 'init'
        workingDirectory: $(work_dir)
        backendServiceArm: $(service_conn)
        backendAzureRmStorageAccountName: 'krishnastg567'
        backendAzureRmContainerName: 'tfstate'
        backendAzureRmKey: 'dev.terraform.tfstate'
    - task: TerraformTask@5
      displayName: Terraform Plan
      condition: eq('${{ parameters.planRequired }}' true)
      inputs:
        provider: 'azurerm'
